$groups = get-localgroup  
foreach ($group in $groups) {
    ""
    "net.exe localgroup ""{0}"" /comment:""{1}"" /add"  -f $group.name, $group.Description
    $members = Get-LocalGroupMember -Group $group.name
    foreach ($member in $members) {
        "net.exe localgroup ""{0}"" ""{1}"" /add" -f   $group.name ,$member.name
    }
}
